9. Write a program to create a Calculator using the switch Statement.

#include<iostream>
using namespace std;
class calculator
{
char op;
  float num1, num2;
public:
int opr(){


	 cout << "Enter operator: +, -, *, /: ";
  cin >> op;

  cout << "Enter two operands:\n";
  cin >> num1 >> num2;

  switch(op) {

    case '+':
      cout << num1 << " + " << num2 << " = " << num1 + num2;
      break;

    case '-':
      cout << num1 << " - " << num2 << " = " << num1 - num2;
      break;

    case '*':
      cout << num1 << " * " << num2 << " = " << num1 * num2;
      break;

    case '/':
      cout << num1 << " / " << num2 << " = " << num1 / num2;
      break;

    default:
      // If the operator is other than +, -, * or /, error message is shown
      cout << "Error! operator is not correct";
      break;
  }
}

};

int main ()
{

calculator obj;
obj.opr();
return 0;
}
